## examples/youcom-research.js — You.com Research API

A complete working example generated by this skill for the You.com `/v1/research` endpoint.

### Code Tab
```javascript
const apiKey        = ai.config.apiKey;
const query         = ai.config.query;
const researchEffort = ai.vars.researchEffort || "standard";

if (!apiKey) throw new Error("Missing API key. Set it in block configuration.");
if (!query)  throw new Error("Missing query. Provide it in block configuration.");

ai.log("Sending research request to You.com...");

const url = "https://api.you.com/v1/research";
const res = await fetch(url, {
  method: "POST",
  headers: {
    "X-API-Key": apiKey,
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    input: query,                    // Note: You.com uses "input" not "query"
    research_effort: researchEffort
  })
});

if (!res.ok) {
  const err = await res.text();
  throw new Error("You.com API error (" + res.status + "): " + err);
}

const data = await res.json();
ai.log("Research complete. Storing results...");

ai.vars[ai.config.researchAnswer] = data.answer ?? data.response ?? JSON.stringify(data);
ai.vars[ai.config.researchRaw]    = JSON.stringify(data);
```

## examples/finnhub-company-profile.js — Finnhub Company Profile API

A complete working example for the Finnhub `/api/v1/stock/profile2` endpoint.
Demonstrates GET request with query param auth and multiple optional lookup fields.

### Code Tab
```javascript
const apiKey = ai.config.apiKey;
const symbol = ai.config.symbol || ai.vars.symbol;
const isin   = ai.config.isin   || ai.vars.isin   || "";
const cusip  = ai.config.cusip  || ai.vars.cusip  || "";

if (!apiKey) throw new Error("Missing API key. Set it in block configuration.");
if (!symbol && !isin && !cusip) throw new Error("Provide at least one of: symbol, ISIN, or CUSIP.");

ai.log("Fetching company profile from Finnhub...");

// Note: URLSearchParams is NOT available in Sandbox — build manually
let query = "?token=" + apiKey;
if (symbol) query += "&symbol=" + encodeURIComponent(symbol);
if (isin)   query += "&isin="   + encodeURIComponent(isin);
if (cusip)  query += "&cusip="  + encodeURIComponent(cusip);

const url = "https://finnhub.io/api/v1/stock/profile2" + query;

try {
  const res = await fetch(url);

  if (!res.ok) {
    const err = await res.text();
    throw new Error("Finnhub error (" + res.status + "): " + err);
  }

  const data = await res.json();

  if (!data || !data.name) {
    throw new Error("No company found for the given symbol/ISIN/CUSIP.");
  }

  ai.log("Profile retrieved. Storing results...");

  ai.vars[ai.config.outputProfile] = JSON.stringify(data);
  ai.vars[ai.config.outputName]    = data.name || "";
  ai.vars[ai.config.outputSummary] =
    data.name + " (" + data.ticker + ") — " + data.finnhubIndustry + ", " + data.country + ". " +
    "Market cap: $" + (data.marketCapitalization ? data.marketCapitalization.toLocaleString() : "N/A") + "M. " +
    "Exchange: " + data.exchange + ". IPO: " + data.ipo + ". Website: " + data.weburl;

} catch (error) {
  ai.log("Error: " + error.message);
  throw error;
}
```
